  code={`// Adam — Kingma & Ba 2014
function adam(grad, x, state, lr=0.001, b1=0.9, b2=0.999, eps=1e-8) {
  state.t = (state.t || 0) + 1
  state.m = b1 * (state.m || 0) + (1 - b1) * grad
  state.v = b2 * (state.v || 0) + (1 - b2) * grad * grad
  const mHat = state.m / (1 - Math.pow(b1, state.t))
  const vHat = state.v / (1 - Math.pow(b2, state.t))
  return x - lr * mHat / (Math.sqrt(vHat) + eps)
}

// Minimize f(x) = x^4 - 3x^3  (non-convex, has saddle near x=0)
let x = 10.0, state = {}, log = []
for (let i = 0; i < 500; i++) {
  const grad = 4 * Math.pow(x, 3) - 9 * Math.pow(x, 2)
  x = adam(grad, x, state)
  if (i % 100 === 0) log.push(\`t=\${i} x=\${x.toFixed(4)}\`)
}
return log.join('\\n') + \`\\nFinal x ≈ \${x.toFixed(6)} (local min near x=2.25)\``}

## NotebookLM Q&A
- **Q:** ¿Por qué Adam converge más rápido que SGD en superficies no-convexas?
  **A:** Adam mantiene estimaciones adaptativas por parámetro: m_t (primer momento, momentum) suaviza la dirección, v_t (segundo momento) escala el learning rate inversamente a la magnitud de los gradientes históricos. Donde el gradiente es ruidoso/pequeño, Adam da pasos proporcionales — SGD da el mismo paso fijo y se atasca o diverge.
